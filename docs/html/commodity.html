Example

cmQuoteMatrix = caplib::CreateCmOptionQuoteMatrix(
    "EUROPEAN", "FUTURE_UNDERLYING_TYPE", 2025.01.02,
    [2025.07.02, 2026.01.02], [["CALL", "PUT"], ["CALL", "PUT"]],

    "AU", "CM_QUOTES", true);
